model {

  alpha ~ dbeta(1,1) # learning rate
  theta1 ~ dbeta(1,1) # initial theta value
  
  theta_learn[1] <- theta1 # initial theta value for the learning process

  for (t in 2:n_trials) {
    
    theta_learn[t] <- theta_learn[t-1]^(1/(1+alpha)) # we keep alpha constant; assuming that it keeps the same through all trials
    Glearn[t] ~ dbin(theta_learn[t], 1)

  }  

}
